\section*{Objectives}

\subsection*{Technical Safety}

\begin{enumerate}
    \item \textbf{Scanning the website}
    \begin{itemize}
        \item \textbf{Goal:} Getting website technical information to evaluate website reliability.
        \item \textbf{Technical details:}
        \begin{itemize}
            \item Scan for server’s IP, protocol security, domain age, domain pattern, and certificate details.
            \item Integrate with trusted APIs such as \textbf{Google AI studio} to check for \textbf{"evil kerning"}\footnote{https://office-watch.com/2025/evil-kerning-phishing-scam/} or typo-based phishing.
            \item Utilize \textbf{reputation databases} and \textbf{user feedback} to enhance detection accuracy.
        \end{itemize}
    \end{itemize}

    \item \textbf{Evaluate website reliability (Trust Score System)}
    \begin{itemize}
        \item \textbf{Goal:} Calculate trust score for the website based on multiple criteria.
        \item \textbf{Technical details:}
        \begin{itemize}
            \item Use features collected from website scanning and external reputation sources.
            \item Assign a numerical trust score (0.0-5.0) indicating safety level.
        \end{itemize}
    \end{itemize}
\end{enumerate}

\subsection*{Content Safety}

\begin{enumerate}
    \item \textbf{Analyze and provide suggestions}
    \begin{itemize}
        \item \textbf{Goal:} Suggest alternative trustworthy websites when a malicious one is detected.
        \item \textbf{Technical details:}
        \begin{itemize}
            \item Determine the website's topic (e.g., shopping, banking, news) by analyzing it's content(text, image, name of website) via \textbf{Gemini} API.
            \item Warn the user when the website contains NSFW\footnote{Not Safe For Work} content. 
            \item Use \textbf{Gemini} API for similar website recommendation.
            \item Notify users via popup with 3–5 suggested sites.
        \end{itemize}
    \end{itemize}
\end{enumerate}
